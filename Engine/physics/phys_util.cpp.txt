#include "phys_util.h"


double PhysUtil::pixMeter = 10.0000000f; // Pixels per meter;


double PhysUtil::met2pix(double met)
{
    return met * pixMeter;
}

double PhysUtil::pix2met(double pix)
{
    return pix / pixMeter;
}
